{

            perror("socket failed");

                            SO_REUSEADDR | SO_REUSEPORT, &opt,

            perror("setsockopt");

                    sizeof(address))
            < 0) {
            perror("bind failed");
            exit(EXIT_FAILURE);
    }
    if (listen(server_fd, 3) < 0) {
            perror("listen");
            exit(EXIT_FAILURE);
    }
    if ((new_socket
            = accept(server_fd, (struct sockaddr*)&address,

    printf("Hello message sent\n");

    // closing the connected socket
    close(new_socket);
    // closing the listening socket
    close(server_fd);
    return 0;
}
