to_uppercase:

  ; rdi = addr of character
  cmp byte [rdi], 'a'
  jb .done

  cmp byte [rdi], 'z'
  ja .done

  .done:
  ret

_start:

  ; Read syscall
  mov rax, SYS_read
  mov rdi, SYS_stdin
  mov rsi, buf
  mov rdx, BUFLEN
  syscall

  mov rax, SYS_write
  mov rdi, SYS_stdout
  mov rsi, buf
  syscall

  ; Exit syscall
  mov rax, SYS_exit
  mov rdi, 0
